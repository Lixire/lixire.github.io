#include <stdio.h>
void count_digits(const int num){
    // the array will keep track of the number of occurrences of each digit
    int counter[10] = {0};
    int x = num;
    for(int i = 0; i < x/10; ++i){
        ++counter[x%10];
        x /= 10;
    }   
    for(int j = 0; j < 10 ; ++j){
        printf("The digit %d appeared %d time(s) in %d\n", j , counter[i], num);
    }
}
